}

[a : $u32 = 6]
[b : u32 <- 5]
[c : u32 = b]

c <- 12 + 3;
printf("a = %d | b = %d | c = %d\n", a, $b, $c);
printf("Hello, World!\n");
printf("factorial: %d\n", factorial(a));
printf(("fibonacci: %d\n", fibonacci(a));

[factorial = (n : $u32) -> $u32 {
	scope[0].ret <- if (n > 1) then n * this(n-1) else 1;
}]

[fibonacci = (n: $u32) -> $u32 {
	scope[0].ret <- if (n > 1) then this(n-1) + this(n-2) 
			else if (n == 1) then 1
			else 0;
}]
